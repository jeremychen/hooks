useInterval


import { useEffect } from 'react'

export default function useInterval (cb: GDUI.IFuncNoop, delay = 1000) {
    useEffect(() => {
        const timer = setInterval(() => {
            cb && cb()
        }, delay)
        return () => clearInterval(timer)
    }, [ cb, delay ])
}
